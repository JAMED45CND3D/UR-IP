// ── FORWARD TEST STATE ── let ftPrediction = []; // array of predicted hot digits let ftResults = JSON.parse(localStorage.getItem('syklus_ft') || '[]'); function setFTPrediction(hotDigits, meta) { ftPrediction = hotDigits; const box = document.getElementById('ft-pred-digits'); box.innerHTML = hotDigits.map(d => `
${d}
` ).join(''); document.getElementById('ft-pred-meta').textContent = meta; renderHistory(); } function addResult() { const actual = document.getElementById('ft-actual').value.trim().replace(/\D/g,''); const period = document.getElementById('ft-period').value.trim() || ('auto-'+(ftResults.length+1)); if (actual.length < 4) { alert('Masukkan 4-5 digit hasil!'); return; } if (ftPrediction.length === 0) { alert('Jalankan Analisis dulu!'); return; } const digits = [...actual.padStart(5,'0')].map(Number); const hits = digits.filter(d => ftPrediction.includes(d)); const hitCount = hits.length; const hitPositions = digits.map((d,i) => ftPrediction.includes(d) ? i : -1).filter(x=>x>=0); ftResults.unshift({ period, actual: actual.padStart(5,'0'), digits, hits: hitCount, hitPositions, prediction: [...ftPrediction], ts: Date.now() }); localStorage.setItem('syklus_ft', JSON.stringify(ftResults)); document.getElementById('ft-actual').value = ''; document.getElementById('ft-period').value = ''; renderHistory(); updateScore(); } function renderHistory() { const el = document.getElementById('ft-history'); if (ftResults.length === 0) { el.innerHTML = '
Belum ada hasil
'; return; } el.innerHTML = ftResults.slice(0,20).map(r => { const digs = [...r.actual].map((d,i) => { const hit = r.prediction.includes(parseInt(d)); return `${d}`; }).join(''); const hitClass = r.hits >= 3 ? 'fr-hit' : r.hits >= 1 ? '' : 'fr-miss'; return `
#${r.period}
${digs}
${r.hits}/5 hit
pred: [${r.prediction.join(',')}]
`; }).join(''); } function updateScore() { const total = ftResults.length; const totalDigits = total * 5; const totalHits = ftResults.reduce((s,r) => s+r.hits, 0); const acc = totalDigits > 0 ? (totalHits/totalDigits*100) : 0; // Expected: if predicting N digits out of 10, expect N/10 * 5 hits per number const predSize = ftPrediction.length || 3; const expectedAcc = (predSize/10)*100; document.getElementById('ft-total').textContent = total; document.getElementById('ft-hits').textContent = totalHits; document.getElementById('ft-acc').textContent = acc.toFixed(1)+'%'; document.getElementById('ft-exp').textContent = '~'+expectedAcc.toFixed(0)+'%'; const accEl = document.getElementById('ft-acc'); if (acc > expectedAcc * 1.5) accEl.className = 'st-val ok'; else if (acc < expectedAcc * 0.5) accEl.className = 'st-val err'; else accEl.className = 'st-val warn'; // Verdict if (total >= 10) { const verdict = document.getElementById('ft-verdict'); verdict.style.display = 'block'; const edge = acc - expectedAcc; if (edge > 15) { verdict.innerHTML = `+${edge.toFixed(1)}%Ada edge signifikan di atas random!
Teruskan test — butuh 50+ untuk konfirmasi`; } else if (edge > 5) { verdict.innerHTML = `+${edge.toFixed(1)}%Sedikit di atas random
Belum bisa disimpulkan — lanjutkan test`; } else if (edge < -5) { verdict.innerHTML = `${edge.toFixed(1)}%Di bawah random — pola tidak terbukti
Atau coba ganti prediksi dengan cold digits`; } else { verdict.innerHTML = `${edge.toFixed(1)}%Setara random — tidak ada pola terdeteksi
Hasil ini mendukung hipotesis random`; } } } function clearResults() { if (!confirm('Reset semua data forward test?')) return; ftResults = []; localStorage.removeItem('syklus_ft'); renderHistory(); updateScore(); } // Hook analyze() to also update FT prediction const _origAnalyze = typeof analyze !== 'undefined' ? analyze : null; window.addEventListener('load', () => { const origAnalyze = window.analyze; window.analyze = function() { origAnalyze(); // After analysis, pull hot digits and set as prediction setTimeout(() => { // Read hot digits from the hot-cold panel const hotColdEl = document.getElementById('hot-cold'); if (!hotColdEl) return; const spans = hotColdEl.querySelectorAll('span[style*="ff6847"]'); const hotDigits = [...spans].map(s => parseInt(s.textContent)).filter(d => !isNaN(d)); if (hotDigits.length > 0) { setFTPrediction(hotDigits, `Hot digits dari analisis · tap tab Forward Test`); } }, 100); }; renderHistory(); updateScore(); });
